      var plugins = [];

      for (var i = 0; i < selected.length; i++) {
        plugins.push($(selected[i]).data('name'));
      };

      if (plugins.length > 0) {
        code.push('');

        code.push('$.foo.pull("'+plugins.join('", "')+'");');
      };

      code = code.join("\n");

      $('#head-tags').text(code);
